def b_search(a, x, r):
    
        i = b.bisect_left(a, x)
        if i < len(a) and abs(a[i] - x) < r:
            return i
        else:
            return -1

def getNearPoints2(lat, lng, r):
    size = training_clean.count().lat
    latpoints = []
    i = b_search(trans_vals[0], lat, r)

    while i > -1 and i < size and r > abs(trans_vals[0][i-1] - lat):
        if r > abs(trans_vals[1][i-1] - lng):
            latpoints.append(1 + 10*trans_vals[2][i-1])
        i += 1

    return np.sum(latpoints)

def compare(origin, dest):

    print 'analysis for trip from ' + res['routes'][0]['legs'][0]['start_address'] + ' to ' + res['routes'][0]['legs'][0]['end_address'] + ':\n'
    
    
    distances = []
    routes = []
    durs = []
    for r in range(len(res['routes'])):
        route = []
        for i in res['routes'][r]['legs'][0]['steps']:
            route.append([round(i['end_location']['lat'], 4), round(i['end_location']['lng'], 4)])
        routes.append(route)
        distances.append(res['routes'][r]['legs'][0]['distance']['value'])
        durs.append(res['routes'][r]['legs'][0]['duration']['text'])

    predicts = []
    for r in routes:
        predicts.append(regr.predict(r))

        print 'distance: ' + str(d) + ' meters'

        print
    return routes
