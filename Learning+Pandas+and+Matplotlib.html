year_and_dist = planets[['DiscoveryYear', 'DistFromSunParsec']]

medians = []
years = range(2015)
years = years[1995:]

for y in years:
    year_df = year_and_dist[year_and_dist['DiscoveryYear'] == y]
    medians.append(year_df['DistFromSunParsec'].median())
    y = y + 1
 
s = pd.Series(medians, index=years)
s.plot(kind='bar', color='black')

plt.title('Median distance of exoplanets by year')
plt.ylabel('Distance from sun (parsecs)')
plt.xlabel('Year of discovery')

from mpl_toolkits.basemap import Basemap

map = Basemap(projection='eck4',lat_0=0,lon_0=0,resolution='l')

# draw coastlines, country boundaries, fill continents.
map.drawcoastlines(linewidth=0.5)
map.drawcountries(linewidth=0.5)
map.fillcontinents(color='white',lake_color='lightblue', alpha=0.5)

# draw the edge of the map projection region (the projection limb)
map.drawmapboundary(fill_color='lightblue')

# draw lat/lon grid lines every 30 degrees.
map.drawmeridians(np.arange(0,360,30))
map.drawparallels(np.arange(-90,90,30))

lons = zika['X'].tolist()
lats = zika['Y'].tolist()

x, y = map(lons, lats)
